```python
class Solution:
    def baseNeg2(self, n: int) -> str:
        if n == 0:
            return "0"
        ans = []
        while n != 0:
            now = n & 1
            ans.append(str(now))
            n -= now
            n //= -2
        ans = ans[::-1]
        return "".join(ans)

```

```TypeScript
function baseNeg2(n: number): string {
    if (n === 0) return "0";
    let ans: string[] = [];
    let base = -2;
    while (n !== 0) {
        let now = n & 1;
        ans.push(String(now));
        n -= now;
        n /= base;
    }
    return ans.reverse().join('');
};
```
